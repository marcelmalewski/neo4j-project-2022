# Aplikacja na początkowej stronie powinno wyświetlić panel do wyszukiwania książek po tytule, autorze i gatunkach
http://localhost:5000/books?title=The Great Gatsby&genres=Fiction     ,Romance&author=F. Scott Fitzgerald
# mozna podac parametr sort_by=title, author, year, rating (tylko te są zaimplementowane)
# defaultowe sort desc albo dodac paramter sort_order=... asc|desc

# Aplikacja na początkowej stronie powinno wyświetlić najbardziej popularne książki na podstawie wystawionych ocen
http://localhost:5000/books/popular/5

# Pobranie szczegółów na temat wybranej książki
# tytuł, zdjecie o ile istnieje, opis, data, gatunki, autorów, wydawnictwo, rok wydania
# srednia ocen z iloscia glosow
http://localhost:5000/books/details/11
